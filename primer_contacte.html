Un primer contacte

Intenta interpretar les següents versions del mateix programa.

Versió pseudocodi

1programa
2  const PI = 3,14159
3  var radi és real, àrea és real
4  escriu "Càlcul de l'àrea d'un cercle"
5  escriu "Introduïu el radi: "
6  llegeix radi
7  àrea = PI * radi^2
8  escriu "L'àrea és ", àrea
9fi programa

Versió C

 1#include <stdio.h>
 2#define PI 3.14159
 3
 4int main() {
 5  float radi, area;
 6  printf("Càlcul de l'àrea d'un cercle\n");
 7  printf("Introduïu el radi: ");
 8  scanf("%f", &radi);               /* interpreta entrada com float */
 9  area = PI * radi * radi;
10  printf("L'àrea és %f\n", area);
11}

Versió Java

 1import java.io.*;
 2
 3public class CalculaRadi {
 4  public static void main(String[] args) throws IOException {
 5    System.out.println("Càlcul de l'àrea d'un cercle");
 6    System.out.println("Introduïu el radi:");
 7    BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
 8    String linia = reader.readLine();
 9    float radi = Float.parseFloat(linia);  // conversió de String a float
10    float area =  (float)Math.PI * radi * radi;
11    System.out.println("L'àrea és " + area);
12  }
13}
