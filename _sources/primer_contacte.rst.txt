Intenta interpretar les següents versions del mateix programa.

Versió pseudocodi
=================

.. code-block:: console
    :linenos:

    programa
      const PI = 3,14159
      var radi és real, àrea és real
      escriu "Càlcul de l'àrea d'un cercle"
      escriu "Introduïu el radi: "
      llegeix radi
      àrea = PI * radi^2
      escriu "L'àrea és ", àrea
    fi programa


Versió C
========

.. code-block:: c
    :linenos:

    #include <stdio.h>
    #define PI 3.14159

    int main() {
      float radi, area;
      printf("Càlcul de l'àrea d'un cercle\n");
      printf("Introduïu el radi: ");
      scanf("%f", &radi);               /* interpreta entrada com float */
      area = PI * radi * radi;
      printf("L'àrea és %f\n", area);
    }


Versió Java
===========

.. code-block:: java
    :linenos:

    import java.io.*;

    public class CalculaRadi {
      public static void main(String[] args) throws IOException {
        System.out.println("Càlcul de l'àrea d'un cercle");
        System.out.println("Introduïu el radi:");
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        String linia = reader.readLine();
        float radi = Float.parseFloat(linia);  // conversió de String a float
        float area =  (float)Math.PI * radi * radi;
        System.out.println("L'àrea és " + area);
      }
    }
